## dashboard.ts (new)

```ts
import Chart from 'chart.js/auto';

export class Dashboard {
  constructor(private ctxDelta: HTMLCanvasElement, private ctxSurvival: HTMLCanvasElement) {}

  renderDelta(data: number[][]) {
    new Chart(this.ctxDelta, {
      type: 'line',
      data: {
        labels: data.map((_, i) => i),
        datasets: [{
          label: 'Î”A Trend',
          data: data.map(v => v.reduce((a,b)=>a+b,0)/v.length),
        }]
      }
    });
  }

  renderSurvival(data: number[]) {
    new Chart(this.ctxSurvival, {
      type: 'line',
      data: {
        labels: data.map((_, i) => i),
        datasets: [{
          label: 'Survival Curve',
          data
        }]
      }
    });
  }
}
```
